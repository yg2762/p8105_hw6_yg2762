``` r
  tibble(
    strap_number = 1:5000,
    strap_sample = rerun(5000, boot_sample(weather_df))
  )
```

``` r

  mutate(
    models = map(strap_sample, ~lm(tmax ~ tmin, data = .x)),
    r_results = map(models, broom::glance)) %>%
```
